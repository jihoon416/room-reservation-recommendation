48   def get_data(driver, date): 
49       url = ("http://student.gs.hs.kr/student/" 
50          "well/goodsUse.do?" 
51          "date=%s" 
52          "&site=MAIN&goodsType=SITE" %date) 
53       driver.get(url) 
54       bs = BeautifulSoup(driver.page_source, "html.parser") 
55       date = datetime.datetime.strptime(date, "%Y%m%d") 
56       day = date.isoweekday() 
57       # print(day) 
58    
59       rows = bs.select("#listTable > table > tbody > tr > td.item > span") 
60       rooms = [] 
61       links = [] 
62       for row in rows: 
63           row_links = row.find_all("a") 
64           y = False 
65           yes_links = [] 
66           for link in row_links: 
67               if link.find("span")["class"] == ["useTime", "userTimeY"]:  # 한 방에 승인된 것이 있고 미승인된 것이 있으면 잘못 신청된 경우이므로 배제 
68                   yes_links.append(link) 
69                   y = True 
70           if not y: 
71               for link in row_links: 
72                   links.append(link) 
73           else: 
74               for link in yes_links: 
75                   links.append(link) 
76    
77       for link in links: 
78           popurl = "http://student.gs.hs.kr"+link.get("href") 
79           driver.get(popurl) 
80           bspop = BeautifulSoup(driver.page_source, "html.parser") 
81           temp = bspop.select("#student > ul > li > dl > dd") 
82           loc = 9     # 관련교사 항목이 몇 번째인지 결정 (신청학생들 목록이 없는 경우 때문) 
83           if not temp[7].find_all("li"): 
84               temp.insert(7, ["Dummy"]*384) 
85               loc = 8 
86           else: 
87               temp[7] = temp[7].find_all("li") 
88               for k in range(len(temp[7])): 
89                   temp[7][k] = str(temp[7][k]) 
90                   temp[7][k] = temp[7][k].replace("<li>", "") 
91                   temp[7][k] = temp[7][k].replace("</li>", "") 
92                   temp[7][k] = temp[7][k].replace("\n", "") 
93                   temp[7][k] = re.sub(r"\s\s+", "", temp[7][k]) 
94           if bspop.select("#student > ul > li > dl > dt")[loc].text == "관련교사":  # 관련교사 항목은 삭제해야 데이터가 동일 
95               del temp[9] 
96           for i in [0,1,2,5,9]: 
97               temp[i] = temp[i].text 
98               temp[i] = temp[i].replace("\n", "") 
99               temp[i] = re.sub(r"\s\s+", "", temp[i]) 
100          for i in [3,4,10]: 
101              temp[i] = temp[i].find("span")["title"] 
102          for i in [6,8]: 
103              temp[i] = temp[i].find("xmp") 
104              temp[i] = str(temp[i]) 
105              temp[i] = temp[i].replace("<xmp>", "") 
106              temp[i] = temp[i].replace("</xmp>", "") 
107          state = bspop.select("#hrText")[0].text 
108          state = state.replace("\n", "") 
109          state = re.sub(r"\s\s+", "", state) 
110          # print(popurl+" : "+state) 
111          rooms.append(Room(temp, day, state)) 
112      return rooms

115  def ready_data(data, room_name): 
116      a = data[room_name] 
117      tempx = [] 
118      tempy = [] 
119      for app in a: 
120          start_time = datetime.datetime.strptime(app[3], "%Y%m%d%H%M")     # 사용 시작 시간 
121          app_time = datetime.datetime.strptime(app[10], "%Y-%m-%d %H:%M:%S.%f")  # 신청 시간 
122          early = (start_time-app_time).total_seconds()//60   # 몇 분 전에 신청했는지 
123          num_students = len(app[7])  # 신청 인원 수 
124          if num_students == 384: 
125              continue 
126          day = app[11] 
127          state = app[12]     # 승인 여부 
128          if float(early)>0: 
129              tempx.append(float(early)) 
130              tempy.append(state == "승인" and 1 or 0) 
131      return [tempx, tempy]
